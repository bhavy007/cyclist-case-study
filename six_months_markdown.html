After checking for discrepencies, combining all the data together in a single frame.

six_months <- bind_rows(month6, month7, month8, month9, month10, month11, month12 )

Removing unneccesary columns from our data

six_months <- six_months %>%  
  select(-c(start_lat, start_lng, end_lat, end_lng))

Creating new Date columns for getting different days of rides

six_months$date <- as.Date(six_months$started_at) #The default format is yyyy-mm-dd
six_months$month <- format(as.Date(six_months$date), "%m")
six_months$day <- format(as.Date(six_months$date), "%d")
six_months$year <- format(as.Date(six_months$date), "%Y")
six_months$day_of_week <- format(as.Date(six_months$date), "%A")

Calculating the ride length and saving it as numeric data for performing calculations

six_months$ride_length <- difftime(six_months$ended_at,six_months$started_at)

six_months_v2 %>% 
  mutate(weekday = wday(started_at, label = TRUE)) %>%  #creates weekday field using wday()

  summarise(number_of_rides = n()                       #calculates the number of rides and average duration 
  ,average_duration = mean(ride_length)) %>%                # calculates the average duration

Visualizing Data

Visualizing the number of rides over different months

six_months_v2 %>% 
  mutate(month_list = months(date)) %>%  #creates weekday field using wday()
  group_by(member_casual, month_list) %>%  #groups by usertype and weekday
  summarise(number_of_rides = n()                           #calculates the number of rides and average duration 
            ,average_duration = mean(ride_length)) %>%      # calculates the average duration
  arrange(month_list)    %>%                        # sorts
  ggplot(aes(x = month_list, y = number_of_rides, fill = member_casual)) +geom_col(position = "dodge")
